\newcommand{\fancyromannumeral}[1]{%
  \ifcase#1 \ensuremath{\emptyset}%
  \or I%
  \or I\kern -1.5pt I%
  \or I\kern -1.5pt I\kern -1.5pt I%
  \or I\kern -1pt V%
  \or V%
  \or V\kern -1pt I%
  \or V\kern -1pt I\kern -1.5pt I%
  \or V\kern -1pt I\kern -1.5pt I\kern -1.5pt I%
  \or I\kern -1pt X%
  \or X%
  \or X\kern -1pt I%
  \or X\kern -1pt I\kern -1.5pt I%
  \else \romannumeral{#1}%
  \fi%
}
